create table sample_product_reviews (
   review_no number(8) constraint samplereview_no_pk primary key,
   review_title varchar2(200) not null,
   review_content varchar2(2000) not null,
   product_no number(8) constraint samplereview_productno_fk 
                        references sample_products (product_no),
   user_id varchar2(20) constraint samplereview_userid_fk
                   references sample_users (user_id),
   review_created_date date default sysdate
);
create sequence review_seq nocache;

제품 상세화면에서의 리뷰
1. 리뷰는 상품상세화면에 출력된다.
2. sample_product_reviews 테이블에서 상품번호에 해당하는
   모든 리뷰들을 조회해서 상품상세화면의 하단에 출력한다.

select *
from sample_product_review
where user_id = ?
order by review_no desc

3. 리뷰 입력폼은 로그인한 사용자가 상품상세페이지를 요청했을 때만 
   출력된다.(즉, 로그인하기 전에는 리뷰를 등록할 수 없다.)

if(savedUser == null){
	response.sendRedirect("login.jsp?fail=deny");
}

4. 리뷰목록은 리뷰를 등록한 순서대로 출력된다.

5. 리뷰의 작성자와 로그인한 사용자가 동일한 경우에만 리뷰 삭제버튼이
   출력된다.

리뷰작성자아이디.equals(로그인한 사용자의 아이디){
	삭제버튼
}

6. ReviewDao객체를 획득한다.

7. ReviewDao의 getReviewsByProductNo(상품번호)를 실행해서 해당 상품의 모든 리뷰를 조회한다.
8. 조회된 리뷰목록을 출력한다.

리뷰 등록하기
1. 제목과 내용을 입력하고 등록버튼을 클릭하면,
   insertreview.jsp에 상품번호, 리뷰제목, 리뷰내용, 페이지번호을 제출한다.
2. insertreview.jsp는 로그인한 사용자만 요청할 수 있다.
   로그인이전의 사용자는 "../user/loginform.jsp?fail=deny&job=리뷰등록" url을 응답으로보낸다.
3. insertreview.jsp로 제출된 폼 입력값을 조회한다.
4. Review객체를 생성해서 상품번호, 리뷰제목, 리뷰내용, 작성자아이디를 저장한다.
5. ReviewDao객체를 획득한다.
6. ReviewDao객체의 insertReview(리뷰정보)를 실행해서 등록한다.
7. detail.jsp?no=상품번호&page=페이지번호를 재요청URL로 보낸다.
